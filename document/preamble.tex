\usepackage[
	acronym,
	nonumberlist,
	toc,
	nopostdot,
	nogroupskip
]{glossaries}
\renewcommand*{\glsclearpage}{}
\renewcommand*{\acronymname}{List of Abbreviations}

\newglossarystyle{gloTable}
{%
	% longtable with three columns:
	\renewenvironment{theglossary}%
		{\begin{longtable}{ll}}%
		{\end{longtable}}%
	% no header:
	\renewcommand*{\glossaryheader}{}%
	% no group headings:
	\renewcommand*{\glsgroupheading}[1]{}%
	% main (level 0) entries displayed in a row
	\renewcommand{\glossentry}[2]{%
		% name in bold:
		\glsentryitem{##1}\glstarget{##1}{\textbf{\glossentryname{##1}}} &
		% description in italic
		\glossentrydesc{##1}\tabularnewline%
	}%
	% sub-entries (same as main)
	\renewcommand{\subglossentry}[3]{\glossentry{##2}{##3}}%
	% blank row between groups if nogroupskip=false
	\ifglsnogroupskip%
		\renewcommand*{\glsgroupskip}{}%
	\else
		\renewcommand*{\glsgroupskip}{ & & \tabularnewline}%
	\fi
}
\setglossarystyle{gloTable}

\makeglossaries%
